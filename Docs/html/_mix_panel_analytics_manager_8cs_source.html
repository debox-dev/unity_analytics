203  private Value ObjectToValue(object obj)
204  {
205  if (obj is DateTime dt)
206  {
207  return dt.ToString("yyyy-dd-MM") + "T" + dt.ToString("HH:mm:ss");
208  }
209  switch (obj)
210  {
211  case int i:
212  return new Value(i);
213  case float f:
214  return new Value(f);
215  case bool b:
216  return new Value(b);
217  case string s:
218  return new Value(s);
219  case long l:
220  return new Value(l);
221  case List<object> list:
222  var valueList = list.Select(ObjectToValue);
223  return new Value(valueList);
224  case Dictionary<string, object> dict:
225  var valueDict = new Dictionary<string, Value>();
226  foreach (var pair in dict)
227  {
228  valueDict[pair.Key] = ObjectToValue(pair.Value);
229  }
230  return new Value(valueDict);
231  default:
232  throw new Exception("Type is not supported by mixpanel: " + obj.GetType().ToString());
233  }
